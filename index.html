// Mythikos Forge Inventory Worker // Deploy to Cloudflare Workers const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, PUT, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', }; function json(data, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...CORS, 'Content-Type': 'application/json' }, }); } function err(msg, status = 400) { return json({ error: msg }, status); } // ═══════════════════════════════════════ // ROUTING // ═══════════════════════════════════════ export default { async fetch(request, env) { if (request.method === 'OPTIONS') { return new Response(null, { headers: CORS }); } const url = new URL(request.url); const path = url.pathname; try { if (path === '/ping') return handlePing(env); if (path === '/products' && request.method === 'GET') return handleGetProducts(env); if (path === '/products' && request.method === 'POST') return handleAddProduct(request, env); if (path === '/inventory' && request.method === 'GET') return handleGetInventory(env); if (path === '/transaction' && request.method === 'POST') return handleTransaction(request, env); if (path === '/sync' && request.method === 'POST') return handleSync(request, env); if (path === '/adjust' && request.method === 'POST') return handleAdjust(request, env); if (path === '/pull-orders' && request.method === 'POST') return handlePullOrders(request, env); return err('Not found', 404); } catch (e) { console.error(e); return err('Internal error: ' + e.message, 500); } } }; // ═══════════════════════════════════════ // PING // ═══════════════════════════════════════ async function handlePing(env) { // Verify sheets connection works try { const token = await getGoogleToken(env); return json({ ok: true, message: 'Mythikos Forge Worker online', sheetsConnected: !!token }); } catch (e) { return json({ ok: false, message: 'Worker online but Sheets auth failed: ' + e.message }); } } // ═══════════════════════════════════════ // PRODUCTS // ═══════════════════════════════════════ async function handleGetProducts(env) { const rows = await sheetsRead(env, 'Products', 'A2:I1000'); const products = (rows || []).map(r => ({ sku: r[0] || '', name: r[1] || '', variant: r[2] || '', qrId: r[3] || r[0] || '', sqId: r[4] || '', etsyId: r[5] || '', tiktokId: r[6] || '', threshold: parseInt(r[7]) || 5, active: r[8] !== 'false', })); // Merge with inventory quantities const invRows = await sheetsRead(env, 'Inventory', 'A2:D1000'); const invMap = {}; (invRows || []).forEach(r => { if (r[0]) invMap[r[0]] = parseInt(r[1]) || 0; }); products.forEach(p => { p.qty = invMap[p.sku] ?? 0; }); return json({ products }); } async function handleAddProduct(request, env) { const { product } = await request.json(); if (!product?.sku || !product?.name) return err('SKU and name required'); const now = new Date().toISOString(); // Append to Products sheet await sheetsAppend(env, 'Products', [[ product.sku, product.name, product.variant || '', product.qrId || product.sku, product.sqId || '', product.etsyId || '', product.tiktokId || '', product.threshold || 5, 'true', ]]); // Add to Inventory sheet await sheetsAppend(env, 'Inventory', [[ product.sku, product.qty || 0, now, product.addedBy || 'system', ]]); // Log initial stock as transaction if qty > 0 if ((product.qty || 0) > 0) { await sheetsAppend(env, 'Transactions', [[ now, product.sku, product.name, product.variant || '', 'restock', product.qty, 'local', 'Initial stock', product.addedBy || 'system', ]]); } return json({ ok: true, product }); } // ═══════════════════════════════════════ // INVENTORY // ═══════════════════════════════════════ async function handleGetInventory(env) { const rows = await sheetsRead(env, 'Inventory', 'A2:D1000'); const inventory = (rows || []).map(r => ({ sku: r[0] || '', qty: parseInt(r[1]) || 0, lastUpdated: r[2] || '', updatedBy: r[3] || '', })); return json({ inventory }); } // ═══════════════════════════════════════ // TRANSACTION (sale or restock) // ═══════════════════════════════════════ async function handleTransaction(request, env) { const { transaction, products } = await request.json(); if (!transaction?.sku) return err('Transaction SKU required'); const { sku, name, variant, type, qty, platform, notes, user } = transaction; const now = new Date().toISOString(); // 1. Log to Transactions sheet await sheetsAppend(env, 'Transactions', [[ now, sku, name, variant || '', type, qty, platform || 'manual', notes || '', user || 'unknown' ]]); // 2. Update Inventory sheet quantity const currentQty = await getInventoryQty(env, sku); const newQty = type === 'sale' ? currentQty - qty : currentQty + qty; await updateInventoryQty(env, sku, newQty, now, user); // 3. Sync to platforms const product = (products || []).find(p => p.sku === sku) || {}; const syncResults = {}; if (product.sqId) { syncResults.squarespace = await syncSquarespace(env, product.sqId, newQty); } if (product.etsyId) { syncResults.etsy = await syncEtsy(env, product.etsyId, newQty); } // 4. Log sync results for (const [platform, result] of Object.entries(syncResults)) { await sheetsAppend(env, 'Sync Log', [[ now, sku, platform, result.ok ? 'success' : 'error', result.error || '' ]]); } return json({ ok: true, newQty, syncResults }); } // ═══════════════════════════════════════ // ADJUST (manual correction) // ═══════════════════════════════════════ async function handleAdjust(request, env) { const { sku, name, variant, newQty, reason, notes, user, product } = await request.json(); if (!sku || newQty === undefined) return err('SKU and newQty required'); const now = new Date().toISOString(); const currentQty = await getInventoryQty(env, sku); const delta = Math.abs(newQty - currentQty); // Log transaction await sheetsAppend(env, 'Transactions', [[ now, sku, name || sku, variant || '', 'adjust', delta, 'manual', `${reason || 'Adjustment'}: ${currentQty}→${newQty}. ${notes || ''}`, user || 'unknown' ]]); // Update inventory await updateInventoryQty(env, sku, newQty, now, user); // Sync platforms if (product?.sqId) await syncSquarespace(env, product.sqId, newQty); if (product?.etsyId) await syncEtsy(env, product.etsyId, newQty); return json({ ok: true, newQty }); } // ═══════════════════════════════════════ // SYNC (full product push from PWA) // ═══════════════════════════════════════ async function handleSync(request, env) { const body = await request.json(); if (body.type === 'product_archived') { // Find the row in Products sheet and set Active column to false const rows = await sheetsRead(env, 'Products', 'A2:I1000'); const rowIndex = (rows || []).findIndex(r => r[0] === body.sku); if (rowIndex !== -1) { const sheetRow = rowIndex + 2; await sheetsUpdate(env, `Products!I${sheetRow}`, [['false']]); } return json({ ok: true }); } if (body.type === 'product_updated') { const p = body.product; const rows = await sheetsRead(env, 'Products', 'A2:I1000'); const rowIndex = (rows || []).findIndex(r => r[0] === p.sku); if (rowIndex !== -1) { const sheetRow = rowIndex + 2; await sheetsUpdate(env, `Products!A${sheetRow}:I${sheetRow}`, [[ p.sku, p.name, p.variant || '', p.qrId || p.sku, p.sqId || '', p.etsyId || '', p.tiktokId || '', p.threshold || 5, 'true' ]]); } return json({ ok: true }); } return handleAddProduct( new Request('https://x/products', { method: 'POST', body: JSON.stringify({ product: body.product }), headers: { 'Content-Type': 'application/json' } }), env ); } return json({ ok: true }); } // ═══════════════════════════════════════ // PULL ORDERS (reconcile external sales) // ═══════════════════════════════════════ async function handlePullOrders(request, env) { const { platforms } = await request.json(); const results = {}; if (platforms?.includes('squarespace') && env.SQ_API_KEY) { results.squarespace = await pullSquarespaceOrders(env); } return json({ ok: true, results }); } // ═══════════════════════════════════════ // INVENTORY HELPERS // ═══════════════════════════════════════ async function getInventoryQty(env, sku) { const rows = await sheetsRead(env, 'Inventory', 'A2:B1000'); if (!rows) return 0; const row = rows.find(r => r[0] === sku); return row ? parseInt(row[1]) || 0 : 0; } async function updateInventoryQty(env, sku, newQty, timestamp, user) { // Find the row number for this SKU const rows = await sheetsRead(env, 'Inventory', 'A2:D1000'); if (!rows) return; const rowIndex = rows.findIndex(r => r[0] === sku); if (rowIndex === -1) { // SKU not in inventory sheet yet — append it await sheetsAppend(env, 'Inventory', [[sku, newQty, timestamp, user || 'system']]); return; } // Update existing row (rowIndex is 0-based, sheet rows start at 2) const sheetRow = rowIndex + 2; await sheetsUpdate(env, `Inventory!A${sheetRow}:D${sheetRow}`, [[ sku, newQty, timestamp, user || 'system' ]]); } // ═══════════════════════════════════════ // SQUARESPACE SYNC // ═══════════════════════════════════════ async function syncSquarespace(env, variantId, qty) { if (!env.SQ_API_KEY) return { ok: false, error: 'No API key configured' }; try { const res = await fetch( `https://api.squarespace.com/1.0/commerce/inventory/${variantId}`, { method: 'POST', headers: { 'Authorization': `Bearer ${env.SQ_API_KEY}`, 'Content-Type': 'application/json', }, body: JSON.stringify({ variantId, quantity: qty, isUnlimited: false, }), } ); if (!res.ok) { const text = await res.text(); return { ok: false, error: `SQ ${res.status}: ${text}` }; } return { ok: true }; } catch (e) { return { ok: false, error: e.message }; } } async function pullSquarespaceOrders(env) { if (!env.SQ_API_KEY) return { error: 'No API key' }; try { const res = await fetch( 'https://api.squarespace.com/1.0/commerce/orders?fulfillmentStatus=PENDING', { headers: { 'Authorization': `Bearer ${env.SQ_API_KEY}` } } ); const data = await res.json(); return { ok: true, orderCount: data.result?.length || 0 }; } catch (e) { return { ok: false, error: e.message }; } } // ═══════════════════════════════════════ // ETSY SYNC // ═══════════════════════════════════════ async function syncEtsy(env, listingOfferingId, qty) { if (!env.ETSY_API_KEY) return { ok: false, error: 'No API key configured' }; // listingOfferingId format: "listingId_offeringId" const [listingId, offeringId] = listingOfferingId.split('_'); if (!listingId || !offeringId) return { ok: false, error: 'Invalid Etsy ID format (expected listingId_offeringId)' }; try { const res = await fetch( `https://openapi.etsy.com/v3/application/listings/${listingId}/products/${offeringId}/inventory`, { method: 'PUT', headers: { 'x-api-key': env.ETSY_API_KEY, 'Content-Type': 'application/json', }, body: JSON.stringify({ quantity: qty }), } ); if (!res.ok) { const text = await res.text(); return { ok: false, error: `Etsy ${res.status}: ${text}` }; } return { ok: true }; } catch (e) { return { ok: false, error: e.message }; } } // ═══════════════════════════════════════ // GOOGLE SHEETS API // ═══════════════════════════════════════ async function getGoogleToken(env) { const now = Math.floor(Date.now() / 1000); const header = btoa(JSON.stringify({ alg: 'RS256', typ: 'JWT' })).replace(/=/g, '').replace(/\+/g, '-').replace(/\//g, '_'); const payload = btoa(JSON.stringify({ iss: env.GCP_CLIENT_EMAIL, scope: 'https://www.googleapis.com/auth/spreadsheets', aud: 'https://oauth2.googleapis.com/token', exp: now + 3600, iat: now, })).replace(/=/g, '').replace(/\+/g, '-').replace(/\//g, '_'); const signingInput = `${header}.${payload}`; // Import the private key const privateKeyPem = env.GCP_PRIVATE_KEY.replace(/\\n/g, '\n'); const pemBody = privateKeyPem.replace(/-----BEGIN PRIVATE KEY-----/, '').replace(/-----END PRIVATE KEY-----/, '').replace(/\s/g, ''); const keyBytes = Uint8Array.from(atob(pemBody), c => c.charCodeAt(0)); const cryptoKey = await crypto.subtle.importKey( 'pkcs8', keyBytes.buffer, { name: 'RSASSA-PKCS1-v1_5', hash: 'SHA-256' }, false, ['sign'] ); const encoder = new TextEncoder(); const signatureBuffer = await crypto.subtle.sign('RSASSA-PKCS1-v1_5', cryptoKey, encoder.encode(signingInput)); const signature = btoa(String.fromCharCode(...new Uint8Array(signatureBuffer))).replace(/=/g, '').replace(/\+/g, '-').replace(/\//g, '_'); const jwt = `${signingInput}.${signature}`; // Exchange JWT for access token const tokenRes = await fetch('https://oauth2.googleapis.com/token', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: `grant_type=urn%3Aietf%3Aparams%3Aoauth%3Agrant-type%3Ajwt-bearer&assertion=${jwt}`, }); const tokenData = await tokenRes.json(); if (!tokenData.access_token) throw new Error('Token error: ' + JSON.stringify(tokenData)); return tokenData.access_token; } async function sheetsRead(env, tab, range) { const token = await getGoogleToken(env); const url = `https://sheets.googleapis.com/v4/spreadsheets/${env.SHEET_ID}/values/${encodeURIComponent(tab + '!' + range)}`; const res = await fetch(url, { headers: { Authorization: `Bearer ${token}` } }); const data = await res.json(); if (data.error) throw new Error('Sheets read error: ' + data.error.message); return data.values || []; } async function sheetsAppend(env, tab, rows) { const token = await getGoogleToken(env); const url = `https://sheets.googleapis.com/v4/spreadsheets/${env.SHEET_ID}/values/${encodeURIComponent(tab + '!A1')}:append?valueInputOption=USER_ENTERED&insertDataOption=INSERT_ROWS`; const res = await fetch(url, { method: 'POST', headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ values: rows }), }); const data = await res.json(); if (data.error) throw new Error('Sheets append error: ' + data.error.message); return data; } async function sheetsUpdate(env, range, rows) { const token = await getGoogleToken(env); const url = `https://sheets.googleapis.com/v4/spreadsheets/${env.SHEET_ID}/values/${encodeURIComponent(range)}?valueInputOption=USER_ENTERED`; const res = await fetch(url, { method: 'PUT', headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ values: rows }), }); const data = await res.json(); if (data.error) throw new Error('Sheets update error: ' + data.error.message); return data; }
